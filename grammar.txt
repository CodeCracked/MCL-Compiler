program		: (namespace)*

namespace	: KEYWORD:namespace IDENTIFIER block

function	: KEYWORD:func IDENTIFIER LPAREN (KEYWORD:(int|float) IDENTIFIER (COMMA KEYWORD:(int|float) IDENTIFIER)*)? RPAREN (KEYWORD:return KEYWORD:(int|float))?

block		: (NEWLINE INDENT statement)*

statement	: KEYWORD:(int|float) IDENTIFIER EQUALS expr
			: IDENTIFIER EQUALS expr

expr		: comp-expr ((KEYWORD:AND|OR) comp-expr)*


comp-expr	: KEYWORD:not comp-expr
			: arith-expr ((EQUALS|NOT_EQUALS|LESS|GREATER|LESS_OR_EQUAL|GREATER_OR_EQUAL) arith-expr)*

arith-expr	: factor ((PLUS|MINUS) factor)*

factor		: atom ((MUL|DIV) atom)*

atom		: INT|FLOAT|IDENTIFIER
			: (PLUS|MINUS) factor
			: LPAREN expr RPAREN
			
test = not 5 == 4 and 3 == 3
test = ((not (5 == 4)) and (3 == 3))
