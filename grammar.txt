expr		: KEYWORD:(int|float) IDENTIFIER EQUALS expr
			: comp-expr ((KEYWORD:AND|OR) comp-expr)*


comp-expr	: KEYWORD:not comp-expr
			: arith-expr ((EQUALS|NOT_EQUALS|LESS|GREATER|LESS_OR_EQUAL|GREATER_OR_EQUAL) arith-expr)*

arith-expr	: factor ((PLUS|MINUS) factor)*

factor		: atom ((MUL|DIV) atom)*

atom		: INT|FLOAT|IDENTIFIER
			: (PLUS|MINUS) factor
			: LPAREN expr RPAREN
